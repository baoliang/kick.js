kick.js
=======

sinatra style routing framework

example:

    app.use(function(req, res, next) {
            req.hello = 'hello world';
            next();
    })

    app.get('/', function(req, res, next) {
        res.end(req.hello);
    })

    app.get('/user/:userid', function(req, res, next) {
        res.end(req.params.userid + req.hello);
    })
